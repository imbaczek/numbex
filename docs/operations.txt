API description
===============

format danych: plik csv ("plik delta"):
        start zakresu, koniec zakresu, adres sip, data zmiany

receive updates:
        input: plik delta
        output: True/False
        opis:
                sz -- start zakresu
                kz -- koniec zakresu
                sip -- adres sip
                data -- data zmiany

                for sz, kz, sip, data in csv:
                    z = zakres (sz, kz)
                    wstaw zakres z

                wstaw zakres z:
                    for z' in zakresy pokrywajace sie z z (SELECT):
                        if zakres z pokrywa sie z lewej strony z zakresem z':
                            wydziel z'-z = (sz', sz-1) (UPDATE)
                        elif zakres z pokrywa sie z prawej strony z z':
                            wydziel z'-z = (kz+1, kz') (UPDATE)
                        elif zakres z pokrywa w calosci z':
                            usun z' (DELETE)
			elif zakres z' pokrywa w calosci z:
			    wydziel z'-z = (sz', sz-1) i (kz+1, kz') (UPDATE, INSERT)
                        wstaw do bazy (INSERT)
		-- mo¿liwa optymalizacja: jeœli w pokrywaj¹cym zakresie dane
		-- s¹ identyczne jak w nowym, to nic nie rób

                zakresy pokrywajace sie:
		    -- sa, ka -- start, koniec zakresu
		    -- a -- zakres
                    a = Zakres(sa, ka)
                    SELECT start, koniec
                    FROM zakresy
                    WHERE start BETWEEN sa AND ka
                        OR koniec BETWEEN sa AND ka
			OR sa <= start AND ka >= end
			OR sa >= start AND ka <= end
                
                wydziel:
                    a -- zakres do zmiany
                    b -- zakres nowy
                    UPDATE zakresy
                    SET start = sb, koniec = kb, data = NOW(), ...
                    WHERE start = sa

                wstaw do bazy:
                    INSERT INTO zakresy
                    (start, koniec, data, sip, ...)
                    VALUES (sz, kz, data, sip, ...)

send data:
        input: brak
        output: plik delta
        opis:
                SELECT start, koniec, sip, data
                FROM zakresy;

send updates:
        input: data: data, od której wysy³aæ modyfikacje
        output: plik delta
        opis:
                SELECT start_zakresu, koniec_zakresu, adres_sip, data_zmiany
                FROM zakresy
